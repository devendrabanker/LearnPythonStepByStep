#Goa: This program is to calculate area of a triangle
#Input: 1. Base 2. Height
#Ouput: Area of a triangle 

#Input Example: 1. Base = 32 2.Height = 42
#Output Expectation : 672

Base = float(input("Enter Base: "))
Height = float(input("Enter Height: "))

AreaOfATriangle = (Base * Height) / 2

print(f" Area of a triangle = {AreaOfATriangle} for Base ={Base} and Height = {Height}")
